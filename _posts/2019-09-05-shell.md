### script demo
#### shell脚本标准输出赋值给变量
``` shell
// $(echo 命令行)
key=$(echo ${LINE} | cut -d '=' -f 1) 
```

#### 以#开头的字符串
``` shell
if [[ $LINE == \#* ]];
  then
  echo "$LINE"
fi
```

#### 根据 = 拆分字符串
``` shell
// 相当于是 找第一个 = 截取字符串，获取两边的值
key=$(echo ${LINE} | cut -d '=' -f 1) 
value=$(echo ${LINE} | cut -d '=' -f 2) 

echo $key
echo $value
```

#### 单中括号 与 双中括号

``` shell

1. 单中括号是 bash 的内部命令，等价于 test 命令。
2. 双中括号是 bash 程序的关键字，更加通用。&&、||、<和> 操作符能够正常存在于[[ ]]条件判断结构中。

// 比较字符串 或 数字
if [[ $key1 = $key2 ]]; then
    echo "变量相等"
else
    echo "变量不相等"
fi

// 多分支
if [[ $XcodeVersion == '10.2' ]]; then
    echo "1"
elif [[ $XcodeVersion == '10.1' ]]; then
    echo "2"
else
	echo "3"
fi

// 且
if [[ $key1 == 1 && $key2 == 1 ]]; then
    echo "1"
else
	echo "2"
fi

// 或
if [[ $key1 == 1 || $key2 == 1 ]]; then
    echo "1"
else
	echo "2"
fi

// 非
if [[ ! $key1 == 1 || $key2 == 1 ]]; then
    echo "1"
else
	echo "2"
fi

// 文件系统属性测试
-e 判断对象是否存在（等价于 -a）
-d 判断对象是否存在，并且为目录
-f 判断对象是否存在，并且为常规文件
-L 判断对象是否存在，并且为符号链接
-h 判断对象是否存在，并且为软链接
-s 判断对象是否存在，并且长度不为0
-r 判断对象是否存在，并且可读
-w 判断对象是否存在，并且可写
-x 判断对象是否存在，并且可执行
-O 判断对象是否存在，并且属于当前用户
-G 判断对象是否存在，并且属于当前用户组
-nt 判断file1是否比file2新  [ "/data/file1" -nt "/data/file2" ]
-ot 判断file1是否比file2旧  [ "/data/file1" -ot "/data/file2" ]

// 文件是否存在
if [[ -f "/data/filename" ]]; then
    echo "文件已存在"
else
    echo "文件不存在"
fi

// 文件夹不存在则创建
if [[ ! -d "data/" ]]; then
    mkdir data
else
    echo "文件夹已经存在"
fi
```

#### 从输出中读取变量(param.plist)

``` shell
// 把变量保存到文件中
/usr/libexec/PlistBuddy -c "Add :kdweibo_bundle_id string $pp_main_bundleId" ${WORKSPACE}/param.plist
/usr/libexec/PlistBuddy -c "Add :app_group_id string $APPGroupId" ${WORKSPACE}/param.plist

// 从文件中读取数据，赋值给变量
bundleId=$(/usr/libexec/PlistBuddy -c "print kdweibo_bundle_id" ${WORKSPACE}/param.plist)
title=$(/usr/libexec/PlistBuddy -c "print app_name" ${WORKSPACE}/param.plist)

// 通过 $() 可以把输出赋值给变量
name=$(echo Jimmy)
echo $name
```
